// Простой и более надёжный вариант с HTML-страницей. // Требования: // 1) Установить Node.js // 2) Установить ffmpeg и убедиться, что команда ffmpeg доступна в терминале // 3) Выполнить: // npm init -y // npm i express multer // 4) Запустить: // node server.js // 5) Открыть в браузере: // http://localhost:3000 const express = require('express'); const multer = require('multer'); const { spawn } = require('child_process'); const fs = require('fs'); const path = require('path'); const os = require('os'); const crypto = require('crypto'); const app = express(); const PORT = 3000; const TEMP_DIR = path.join(os.tmpdir(), 'simple-video-compressor'); fs.mkdirSync(TEMP_DIR, { recursive: true }); const upload = multer({ dest: TEMP_DIR, limits: { fileSize: 1024 * 1024 * 1024, // 1 GB }, }); function deleteIfExists(filePath) { if (!filePath) return; fs.unlink(filePath, () => {}); } function escapeHtml(text) { return String(text) .replace(/&/g, '&') .replace(//g, '>') .replace(/"/g, '"') .replace(/'/g, '''); } app.get('/', (req, res) => { res.type('html').send(` Сжатие видео

Сжатие видео

Выбери видеофайл и нажми кнопку. Настройки фиксированные, чтобы всё было проще и стабильнее.

Используется ffmpeg с настройками: H.264, CRF 28, preset medium, AAC 128k.
`); }); app.post('/compress', upload.single('video'), (req, res) => { if (!req.file) { return res.status(400).send('Файл не загружен'); } const originalName = req.file.originalname || 'video'; const safeBaseName = path.parse(originalName).name.replace(/[^a-zA-Z0-9_-]/g, '_') || 'video'; const outputName = `${safeBaseName}_compressed.mp4`; const outputPath = path.join(TEMP_DIR, `${crypto.randomUUID()}_${outputName}`); const ffmpegArgs = [ '-y', '-i', req.file.path, '-c:v', 'libx264', '-preset', 'medium', '-crf', '28', '-c:a', 'aac', '-b:a', '128k', '-movflags', '+faststart', outputPath, ]; const ffmpeg = spawn('ffmpeg', ffmpegArgs); let errorText = ''; ffmpeg.stderr.on('data', (chunk) => { errorText += chunk.toString(); }); ffmpeg.on('error', (err) => { deleteIfExists(req.file.path); deleteIfExists(outputPath); res.status(500).type('html').send(`

Ошибка

Не удалось запустить ffmpeg.

${escapeHtml(err.message)}

Проверь, установлен ли ffmpeg и доступен ли он в PATH.

Назад

`); }); ffmpeg.on('close', (code) => { deleteIfExists(req.file.path); if (code !== 0) { deleteIfExists(outputPath); return res.status(500).type('html').send(`

Ошибка сжатия

ffmpeg завершился с кодом ${code}.

${escapeHtml(errorText)}

Назад

`); } res.download(outputPath, outputName, (downloadError) => { deleteIfExists(outputPath); if (downloadError && !res.headersSent) { res.status(500).send('Не удалось отправить файл'); } }); }); }); app.listen(PORT, () => { console.log(`Открой: http://localhost:${PORT}`); });
